post('/login') do
    if logTime(session[:stress], session[:timeLogged])
        username = params[:username]
        password = params[:password]
        sessions = login(username, password)

        if sessions == false
            redirect('/error')
        else
            session[:id] = sessions[0]
            session[:auth] = sessions[1]
        end
        redirect('/')
    else
        redirect('/showlogin')
    end
end
